===============================================
groovy
===============================================
pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                echo 'Building...'
                sh 'mvn clean compile'
            }
        }

        stage('Test') {
            steps {
                echo 'Testing...'
                sh 'mvn test'
            }
        }

        stage('Package') {
            steps {
                echo 'Packaging...'
                sh 'mvn package'
            }
        }
    }

    post {
        success {
            echo 'Build, Test, and Package stages completed successfully.'
        }
        failure {
            echo 'One of the stages failed. Please check the logs.'
        }
    }
}
